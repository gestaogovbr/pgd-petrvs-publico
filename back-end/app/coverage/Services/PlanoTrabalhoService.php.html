477    {

488                return date("Y-m-d", strtotime($data . " + " . ($dayWeek < $programa->periodicidade_valor ? 7 + $programa->periodicidade_valor - $dayWeek : 7 + 6 - $dayWeek + $programa->periodicidade_valor) . " days"));
489                break;
490            case 'MENSAL':
491                $incMonth = 0;
492            case 'BIMESTRAL':
493                $incMonth = 1;
494            case 'TRIMESTRAL':
495                $incMonth = 2;
496            case 'SEMESTRAL':
497                $incMonth = 5;
498            default:
499                $incMonth = 0;
500        }
501        /* Calulo para MENSAL, BIMESTRAL, TRIMESTRAL e SEMANAL */
502        $ano = date('Y', strtotime($data)); /* 20xx */
503        $mes = date('m', strtotime($data)); /* 01 - 12 */
504        $dia = date('d', strtotime($data)); /* 01 - 12 */
505        if ($dia >= $programa->periodicidade_valor)

507        $anoMesDia = date("Y-m-d", strtotime($ano . "-" . $mes . "-01 + " . $incMonth . " month"));
508        $days = min(date('t', strtotime($anoMesDia)), $programa->periodicidade_valor) - 1;
509        return date("Y-m-d", strtotime($anoMesDia . " + " . $days . " days"));
510    }

516     */
517    public function atualizaConsolidacoes($plano)
518    {
519        DB::transaction(function () use ($plano) {
520            $existentes = $plano->consolidacoes->sortBy('data_inicio')->values();
521            $merged = [];
522            $dataInicioVigencia = Carbon::parse($plano->data_inicio)->toDateString(); /* Transforma de datetime para date */
523            $dataFimVigencia = Carbon::parse($plano->data_fim)->toDateString(); /* Transforma de datetime para date */
524            $dataInicio = Carbon::parse($dataInicioVigencia);
525            $limite = Carbon::parse($dataFimVigencia);
526            while ($dataInicio->lessThanOrEqualTo($limite)) {
527                $proximoFim =  Carbon::parse($this->proxDataConsolidacao($dataInicio->toDateString(), $plano->programa));
528                $dataFim = $proximoFim->greaterThan($limite) ? $limite->copy() : $proximoFim;

530                $intersecao = $existentes->first(fn($c) => $c->status !== "INCLUIDO" && $dataInicio->lessThanOrEqualTo(Carbon::parse($c->data_fim)) && $dataFim->greaterThanOrEqualTo(Carbon::parse($c->data_inicio)));
531                if (!empty($igual)) { /* (RN_CSLD_4) Caso exista perodos iguais, o perodo existente ser mantido (para este perdo nada ser feito, manter a mesma ID) */
532                    $merged[] = $igual;
533                    $existentes = $existentes->reject(fn($e) => $e->id === $igual->id)->values();
534                    $dataInicio = Carbon::parse($igual->data_fim)->addDay();
535                } else if (!empty($intersecao)) { /* Se houver interseco do perodo gerado com um existente que esteja com status CONCLUIDO ou AVALIADO */
536                    $existentes = $existentes->reject(fn($e) => $e->id === $intersecao->id)->values();
537                    if ($intersecao->data_inicio === $dataInicio->toDateString()) { /* (RN_CSLD_5) Se as datas de incio forem iguais o periodo existente ser mantido */
538                        $dataInicio = Carbon::parse($intersecao->data_fim)->addDay();
539                    } else { /* (RN_CSLD_6) Se as datas de incio forem diferente, ento ser criado um novo perdo entre o novo incio e o incio do perodo CONCLUIDO/AVALIADO, e o perodo CONCLUIDO/AVALIADO ser mantido */
540                        $novo = new PlanoTrabalhoConsolidacao([
541                            'data_inicio' => $dataInicio->toDateString(),
542                            'data_fim' =>  Carbon::parse($intersecao->data_inicio)->subDay()->toDateString(),
543                            'plano_trabalho_id' => $plano->id,
544                            'status' => 'INCLUIDO'
545                        ]);
546                        $novo->save();
547                        $merged[] = $novo;
548                        $dataInicio =  Carbon::parse($intersecao->data_inicio);
549                    }
550                } else { /* Novo perodo */
551                    $novo = new PlanoTrabalhoConsolidacao([
552                        'data_inicio' => $dataInicio->toDateString(),
553                        'data_fim' => $dataFim->toDateString(),
554                        'plano_trabalho_id' => $plano->id,
555                        'status' => 'INCLUIDO'
556                    ]);
557                    $novo->save();
558                    $merged[] = $novo;
559                    $dataInicio = $dataFim->copy()->addDay();
560                }
561            }
562            foreach ($existentes as $anterior) {
563                /* Remove o registro da consolidao completamente vazio */
564                $anterior->delete();
565            }
566            /* Refaz todas as datas finais das consolidaes considerando sempre data_inicio da prxima - 1 dia */
567            $this->atualizaDataFimConsolidacoes($plano->id);
568        });
569    }
